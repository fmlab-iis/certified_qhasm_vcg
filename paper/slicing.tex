
\begin{algorithm}
  \begin{algorithmic}[1]
    \Match{$e$}
      \Case{$V$}
        \Return $\{ e \}$
      \EndCase
      \Case{$Z$}
        \Return $\emptyset$
      \EndCase
      \Case{$-e'$}
        \Return \textit{VarsInE} ($e'$)
      \EndCase
      \Case{$e' + e''$}
        \Return \textit{VarsInE} ($e'$) $\cup$ \textit{VarsInE} ($e''$)
      \EndCase
      \Case{$e' - e''$}
        \Return \textit{VarsInE} ($e'$) $\cup$ \textit{VarsInE} ($e''$)
      \EndCase
      \Case{$e' \times e''$}
        \Return \textit{VarsInE} ($e'$) $\cup$ \textit{VarsInE} ($e''$)
      \EndCase
      \Case{$\dslcode{Pow}(e', \uscore)$}
        \Return \textit{VarsInE} ($e'$)
      \EndCase
    \EndMatch
  \end{algorithmic}
  \caption{\textit{VarsInE} ($e$)}
\end{algorithm}

\begin{algorithm}
\begin{algorithmic}[1]
  \Match{$q$}
    \Case{$\top$}
      $\emptyset$
    \EndCase
    \Case{$e' = e''$}
      \Return \textit{VarsInE} ($e'$) $\cup$ \textit{VarsInE} ($e''$)
    \EndCase
    \Case{$e' \equiv e'' \mod \uscore$}
      \Return \textit{VarsInE} ($e'$) $\cup$ \textit{VarsInE} ($e''$)
    \EndCase
    \Case{$q' \wedge q''$}
      \Return \textit{VarsInQ} ($q'$) $\cup$ \textit{VarsInQ} ($q''$)
    \EndCase
  \EndMatch
\end{algorithmic}
\caption{\textit{VarsInQ} ($q$)}
\end{algorithm}

\begin{algorithm}
  \begin{algorithmic}[1]
    \Match{$s$}
      \Case{$v \leftarrow e$}
        \IfThenElse{$v \in \mathit{vars}$}
          {\Return \textit{VarsInE} ($e$) $\cup$ $(\mathit{vars} \setminus \{
            v \})$, $s$}
          {\Return $\mathit{vars}$}
      \EndCase
      \Case{$v_h, v_l \leftarrow \dslcode{Split} (e, \uscore)$}
        \IfThenElse{$v_h \in \mathit{vars}$ \textbf{or} $v_l \in \mathit{vars}$}
          {\Return \textit{VarsInE} ($e$) $\cup$ $(\mathit{vars} \setminus \{
            v_h, v_h \})$, $s$}
          {\Return $\mathit{vars}$}
      \EndCase
    \EndMatch
  \end{algorithmic}
  \caption{\textit{SliceStatement} ($\mathit{vars}$, $s$)}
\end{algorithm}

\begin{algorithm}
  \begin{algorithmic}[1]
    \Match{$p$}
      \Case{$\epsilon$}
        \Return $\epsilon$
      \EndCase
      \Case{$pp; s;$}
        \Match{\textit{SliceStatement} ($\mathit{vars}$, $s$)}
          \Case{$\mathit{vars}'$}
            \Return \textit{SliceProgram} ($\mathit{vars}'$, $pp;$)
          \EndCase
          \Case{$\mathit{vars}'$, $s'$}
            \Return \textit{SliceProgram} ($\mathit{vars}'$, $pp$)$\ s';$
          \EndCase
        \EndMatch
      \EndCase
    \EndMatch
  \end{algorithmic}
  \caption{\textit{SliceProgram} ($\mathit{vars}$, $p$)}
\end{algorithm}

\begin{theorem}
  For every $q, q' \in Q$ and $p \in P$,
  \begin{center}
  $\cond{q}$ $p$ $\cond{q'}$ if and only if
  $\cond{q}$ 
  $\textmd{\textit{SliceProgram}}(\textmd{\textit{VarsInQ}}(q'), p)$
  $\cond{q'}$.
  \end{center}

\end{theorem}
